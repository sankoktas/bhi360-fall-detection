project(boards C)

if(BOARDS STREQUAL "")
	message(FATAL_ERROR "No boards defined for Distribution ${DIST_TYPE}")
endif()

foreach(b ${BOARDS} ${BOARDS_DUT})
	set(__path ${CMAKE_CURRENT_LIST_DIR}/${b}.cfg)
	if(NOT EXISTS ${__path})
		message(FATAL_ERROR "board file ${__path} does not exist")
	endif()
endforeach()
